/*1. Dado um número de 3 algarismos N = CDU (onde C é o algarismo das centenas, D é o
algarismo das dezenas e U o algarismo das unidades), retorne a decomposição deste número.
Ex: N=321 (3 centenas, 2 dezenas e 1 unidade)*/

var N,C,D,U;

N=Number(prompt("Digite um número de 3 algarismos: "));

C=Math.floor(N/100);
D=Math.floor((N%100)/10);
U=Math.floor((N%100)%10);

console.log(C + " Centenas, " + D + " Dezenas e " + U + " Unidades");
