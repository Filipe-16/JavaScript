/*1. Calcule a área de um terreno retangular, a partir das dimensões do terreno.*/

var d1,d2;

d1=Number(prompt("Digite a dimensão 1 do terreno:"));
d2=Number(prompt("Digite a dimensão 2 do terreno:"));

console.log("A área do terreno retangular é: " + d1*d2);

/*2. Calcule quantos dias de vida tem uma pessoa. O algoritmo deve ler o nome e a idade da
pessoa (considerando sempre anos completos). Ex: uma pessoa com 15 anos tem 5.475 dias.*/

var nome,idade;

nome=prompt("Digite o nome:");
idade=Number(prompt("Digite a idade (considerando anos completos):"));

console.log(nome + " com " + idade + " anos tem " + idade*365 + " dias");

/*3. Leia uma temperatura em graus Celsius (C) e apresente-a convertida em graus Farenheit (F).*/

var C,F;

C=Number(prompt("Digite uma temperatura em graus Celsius: "));

F=C*1.8+32;

console.log("A temperatura em graus Farenheit é: " + F);

/*4. Leia uma temperatura em graus Farenheit (F) e apresente-a convertida em graus Celsius (C).*/

var F,C;

F=Number(prompt("Digite uma temperatura em graus Farenheit: "));

C=(F-32)/1.8;

console.log("A temperatura em graus Celsius é: " + C);

/*5. Calcule e apresente o valor do volume de uma lata de óleo.*/

var r,h,v;

r=Number(prompt("Digite o raio:"));
h=Number(prompt("Digite a altura:"));

v=3.14*(r**2)*h;

console.log("Volume da lata de óleo: " + v);

/*6. Calcular a média ponderada de três notas de um aluno (as notas têm pesos 1, 2 e 7,
respectivamente).*/

var n1,n2,n3,mp;

n1=Number(prompt("Digite a nota 1:"));
n2=Number(prompt("Digite a nota 2:"));
n3=Number(prompt("Digite a nota 3:"));

mp=(n1*1+n2*2+n3*7)/10;

console.log("A média ponderada é: " + mp);

/*7. Efetue o cálculo e a apresentação do valor de uma prestação em atraso, onde o valor original
da prestação, a taxa mensal de juros e o tempo de atraso (em meses) deve ser informado pelo
usuário.*/

var vp,tj,tam,pa;

vp=Number(prompt("Digite o valor original da prestação: "));
tj=Number(prompt("Digite a taxa mensal de juros: "));
tam=Number(prompt("Digite o tempo de atraso (em meses): "));

pa=vp + (vp/100*tj);

console.log("O valor de uma prestação em atraso é: " + pa);

/*8. Ler o peso de uma pessoa. Calcular e mostrar:
- o novo peso se a pessoa engordar 15% sobre o peso digitado;
- o novo peso se a pessoa emagrecer 20% do peso digitado.*/

var p,pg,pe;

p=Number(prompt("Digite o peso da pessoa: "));

pg=p+(p*0.15);
pe=p-(p*0.20);

console.log("Peso engordado: " + pg + " peso emagrecido: " + pe);

/*9. Apresente o valor da conversão em real (R$) de um valor lido em dólar (US$).*/

var d,r;

/*11. Leia dois valores para as variáveis A e B, e efetue a troca dos valores de forma que a variável
A passe a possuir o valor da variável B e a variável B passe a possuir o valor da variável A.
Apresentar os valores trocados.*/

a=Number(prompt("Digite um valor para A:"));

aux=a;
a=b;
b=aux;

/*12. Leia dois valores inteiros (variáveis A e B) e apresente o resultado do quadrado da diferença

/*13. Leia três valores inteiros (representados pelas variáveis A, B e C) e apresente como resultado
