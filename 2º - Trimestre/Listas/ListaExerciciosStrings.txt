◦ book = books
◦ girl = girls

◦ tomato = tomatoes
◦ church = churches
◦ brush = brushes
◦ glass = glasses
◦ box = boxes

  console.log(p + "es");

//Palavras terminadas em 'y', precedidos de consoante

  console.log(p.slice(0,p.length-1) + "ies");
}
else{
    console.log(p + "s");
}

  if(s.charCodeAt(i)>=97 && s.charCodeAt(i)<=122){
    l+=s.charAt(i);
  }

  console.log("Não é um palíndromo");
