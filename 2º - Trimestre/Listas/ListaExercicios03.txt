/*2. Calcular o somatório, a soma dos quadrados e a média entre os n primeiros números inteiros
positivos.*/

var n,s,q,m,i;

s=0;
q=0;

n=Number(prompt("Digite os primeiros números inteiros positivos: "));

for(i=1;i<=n;i++){
  s+=i;
  q+=i**2;
}
m=s/n;

console.log("O somatório é: " + s + " a soma dos quadrados é: " + q + " e a média é: " + m);

/*5. Calcular o faturamento total mensal de um armazém, sabendo-se que, para cada uma das
mercadorias com que o armazém trabalha dispõe-se dos seguintes dados:
− o nome da mercadoria;
− o seu preço unitário;
− a quantidade total vendida no mês.
O faturamento total do armazém será calculado somando-se o total faturado por cada
mercadoria, que é igual a quantidade vendida da mercadoria vezes o seu preço unitário. O
número de mercadorias comercializadas pelo armazém deve ser informado pelo usuário.*/

var nmc,nm,pu,qvm,fpm,fta,i;

fta=0;

nmc=Number(prompt("Digite o número de mercadorias comercializadas pelo armazém: "));

for(i=0;i<nmc;i++){
  nm=prompt("Digite o nome da mercadoria: ");
  pu=Number(prompt("Digite o preço unitário: "));
  qvm=Number(prompt("Digite a quantidade total vendida no mês: "));

  fpm=qvm*pu;
  fta+=fpm;
}
console.log("O faturamento total do armazém é: " + fta);

/*11. Calcular N! (fatorial de N), sendo que o valor de N é fornecido pelo usuário e sabendo que:

N! = 1 x 2 x 3 x ... x (N - 1) x N
0! = 1, por definição*/

var n,i,f;

n=Number(prompt("Digite um número:"))

f=1
    
if(n%1==0 && n>-1){
    for(i=1;i<n;i++){
        f*=i+1
    }
    console.log("o fatorial de n é: " + f)
}
else{
    console.log("Não existe fatorial para este número") 
}
